1-A. マージ済みブランチの一覧を出す
bash

例:

bash

origin/hotfix/tmp-fix-20250728
1-B. 不要ブランチを削除
bash


# 1 本ずつ
git push origin --delete feature/supercronic-chaos

# まとめ削除 (安全のため dry-run 推奨)
for b in $(git branch -r --merged origin/main | \
           grep -vE 'origin/(main|develop|HEAD)' | sed 's|origin/||'); do
  echo "Would delete $b"
done

# 確認後に:
for b in $(git branch -r --merged origin/main | \
           grep -vE 'origin/(main|develop|HEAD)' | sed 's|origin/||'); do
  git push origin --delete "$b"
done
